# square-and-cube-of-a-number


#include <stdio.h>
int main()
{
  float x,square_result,cube_result;
  
  printf("enter x\n");
  scanf("%f" , &x);
  
  
  square_result=x*x;
  
  printf("square of x %f\n"  , square_result);
  
  
  cube_result=x*x*x;
  
  printf("cube of x %f\n" ,cube_result );
}
